Ennustamisen kohteena on Suomen kuluttajintainflaation vuosimuotos eli HICP. Datan kerään Eurostatilta, joka on Euroopan unionin tilastokeskus.


```{r echo = FALSE, message=FALSE}
# Paketointi
library(eurostat)
library(dplyr)
library(tsibble)
library(ggplot2)
library(lubridate)
library(feasts)

# HICP kuukausi-indeksi
hicp_raw <- get_eurostat(
  "prc_hicp_midx",
  time_format = "date"
)

# Suomi, kokonaisindeksi
hicp_fi <- hicp_raw |>
  filter(
    geo == "FI",
    coicop == "CP00",
    unit == "I15"
  ) |>
  select(time = TIME_PERIOD, values) |>
  arrange(time)

# Muodostetaan tsibble
hicp_ts <- hicp_fi |>
  mutate(time = yearmonth(time)) |>   # <- kriittinen muunnos
  as_tsibble(index = time)

# Lasketaan vuosimuutos (%)
hicp_yoy <- hicp_ts |>
  mutate(
    inflation_yoy = 100 * (values / lag(values, 12) - 1)
  ) |>
  filter(!is.na(inflation_yoy))

```

hicp_yoy |>
  mutate(year = year(as.Date(time))) |>
  ggplot(aes(x = month(as.Date(time), label = TRUE),
             y = inflation_yoy,
             group = year)) +
  geom_line(
    data = ~ filter(.x, year < max(year) - 2),
    colour = "grey80",
    linewidth = 0.6
  ) +
  geom_line(
    data = ~ filter(.x, year >= max(year) - 2),
    aes(colour = factor(year)),
    linewidth = 1
  ) +
  scale_colour_manual(values = c("#1b9e77", "#d95f02", "#7570b3")) +

    title = "Suomen inflaation kausiprofiili",
    subtitle = "Viimeiset kolme vuotta korostettu",
    x = NULL,
    y = "Inflaatio (%)",
    colour = "Vuosi"
